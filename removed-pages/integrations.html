🔗 Quick API Example
// Get player performance data
// Initialize the Blaze Intelligence API client
const blaze = new BlazeAPI({
  apiKey: 'your-api-key',
  endpoint: 'https://api.blaze-intelligence.com'
});

// Fetch player stats for the season
const playerStats = await blaze.players.getStats({
  playerId: '12345',
  season: '2024',
  metrics: ['points', 'assists', 'efficiency']
});

// Get real-time game predictions
const predictions = await blaze.games.getPredictions({
  gameId: 'game-67890',
  includePlayerImpact: true
});
