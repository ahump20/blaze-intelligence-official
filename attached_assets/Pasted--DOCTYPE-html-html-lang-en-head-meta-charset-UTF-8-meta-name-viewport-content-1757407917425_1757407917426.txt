    <script>
    document.addEventListener('DOMContentLoaded', () => {
        const scenes = [];

        function setupScene(canvasId) {
            const canvas = document.getElementById(canvasId);
            if (!canvas) return null;

            const renderer = new THREE.WebGLRenderer({ canvas, alpha: true, antialias: true });
            renderer.setPixelRatio(window.devicePixelRatio);
            
            const scene = new THREE.Scene();
            const camera = new THREE.PerspectiveCamera(75, canvas.clientWidth / canvas.clientHeight, 0.1, 1000);
            camera.position.z = 5;

            const sceneData = { canvas, renderer, scene, camera, objects: [] };
            scenes.push(sceneData);
            return sceneData;
        }

        // --- Hero Scene ---
        function initHeroScene() {
            const sceneData = setupScene('hero-canvas');
            if (!sceneData) return;
            
            const { scene } = sceneData;
            
            const geometry = new THREE.BufferGeometry();
            const vertices = [];
            for (let i = 0; i < 500; i++) {
                vertices.push(
                    (Math.random() - 0.5) * 30,
                    (Math.random() - 0.5) * 20,
                    (Math.random() - 0.5) * 10
                );
            }
            geometry.setAttribute('position', new THREE.Float32BufferAttribute(vertices, 3));
            
            const material = new THREE.PointsMaterial({
                color: getComputedStyle(document.documentElement).getPropertyValue('--burnt-orange'),
                size: 0.05,
                transparent: true,
                blending: THREE.AdditiveBlending
            });
            
            const particles = new THREE.Points(geometry, material);
            scene.add(particles);
            sceneData.objects.push(particles);
        }

        // --- ShowIQ Scene ---
        function initShowIQScene() {
            const sceneData = setupScene('showiq-canvas');
            if (!sceneData) return;

            const { scene, objects } = sceneData;
            sceneData.camera.position.z = 4;
            
            const geometry = new THREE.IcosahedronGeometry(1.5, 0);
            const material = new THREE.MeshStandardMaterial({
                color: getComputedStyle(document.documentElement).getPropertyValue('--cardinal-blue'),
                wireframe: true,
                metalness: 0.5,
                roughness: 0.5
            });
            const mesh = new THREE.Mesh(geometry, material);
            scene.add(mesh);
            objects.push(mesh);
            
            const light = new THREE.PointLight(0xffffff, 1, 100);
            light.position.set(0, 0, 5);
            scene.add(light);
        }
        
        // --- Pressure Index Scene ---
        function initPressureScene() {
            const sceneData = setupScene('pressure-canvas');
            if (!sceneData) return;
            const { scene, objects } = sceneData;

            const group = new THREE.Group();
            const centerSphere = new THREE.Mesh(
                new THREE.SphereGeometry(0.5, 32, 16),
                new THREE.MeshBasicMaterial({ color: getComputedStyle(document.documentElement).getPropertyValue('--burnt-orange') })
            );
            group.add(centerSphere);

            for (let i = 0; i < 5; i++) {
                const mesh = new THREE.Mesh(
                    new THREE.BoxGeometry(0.3, 0.3, 0.3),
                    new THREE.MeshStandardMaterial({ color: getComputedStyle(document.documentElement).getPropertyValue('--cardinal-blue'), metalness: 0.7 })
                );
                const angle = (i / 5) * Math.PI * 2;
                mesh.position.set(Math.cos(angle) * 1.5, Math.sin(angle) * 1.5, 0);
                group.add(mesh);
            }
            scene.add(group);
            objects.push(group);
            const light = new THREE.PointLight(0xffffff, 1, 100);
            light.position.set(0, 0, 5);
            scene.add(light);
        }

        // --- Clutch Grade Scene ---
        function initClutchScene() {
             const sceneData = setupScene('clutch-canvas');
            if (!sceneData) return;
            const { scene, objects } = sceneData;

            const material = new THREE.MeshStandardMaterial({ color: getComputedStyle(document.documentElement).getPropertyValue('--platinum'), metalness: 0.8, roughness: 0.2 });
            const crossbar = new THREE.Mesh(new THREE.CylinderGeometry(0.1, 0.1, 4, 8), material);
            crossbar.rotation.z = Math.PI / 2;
            
            const post1 = new THREE.Mesh(new THREE.CylinderGeometry(0.1, 0.1, 2, 8), material);
            post1.position.set(-2, 1, 0);
            const post2 = new THREE.Mesh(new THREE.CylinderGeometry(0.1, 0.1, 2, 8), material);
            post2.position.set(2, 1, 0);
            
            const group = new THREE.Group();
            group.add(crossbar, post1, post2);
            group.scale.set(0.6, 0.6, 0.6);
            scene.add(group);
            objects.push(group);

            const light = new THREE.DirectionalLight(0xffffff, 0.8);
            light.position.set(1, 1, 1);
            scene.add(light);
        }

        function resizeRendererToDisplaySize(sceneData) {
            const { renderer, canvas } = sceneData;
            const width = canvas.clientWidth;
            const height = canvas.clientHeight;
            const needResize = canvas.width !== width || canvas.height !== height;
            if (needResize) {
                renderer.setSize(width, height, false);
            }
            return needResize;
        }
        
        const clock = new THREE.Clock();

        function animate() {
            requestAnimationFrame(animate);
            const delta = clock.getDelta();

            scenes.forEach(sceneData => {
                if (resizeRendererToDisplaySize(sceneData)) {
                    const { canvas, camera } = sceneData;
                    camera.aspect = canvas.clientWidth / canvas.clientHeight;
                    camera.updateProjectionMatrix();
                }

                if (sceneData.canvas.id === 'hero-canvas') {
                    sceneData.objects[0].rotation.y += delta * 0.05;
                }
                if (sceneData.canvas.id === 'showiq-canvas') {
                    sceneData.objects[0].rotation.x += delta * 0.2;
                    sceneData.objects[0].rotation.y += delta * 0.2;
                }
                 if (sceneData.canvas.id === 'pressure-canvas') {
                    sceneData.objects[0].rotation.z += delta * 0.1;
                    const time = clock.getElapsedTime();
                    sceneData.objects[0].children.forEach((child, i) => {
                        if (i > 0) { // Not the center sphere
                            const angle = (i / 5) * Math.PI * 2 + time * 0.5;
                            const radius = 1.5 + Math.sin(time + i) * 0.2;
                            child.position.set(Math.cos(angle) * radius, Math.sin(angle) * radius, 0);
                        }
                    });
                }
                if (sceneData.canvas.id === 'clutch-canvas') {
                    sceneData.objects[0].rotation.y += delta * 0.3;
                }
                
                sceneData.renderer.render(sceneData.scene, sceneData.camera);
            });
        }
        
        initHeroScene();
        initShowIQScene();
        initPressureScene();
        initClutchScene();
        animate();
    });
    </script>
